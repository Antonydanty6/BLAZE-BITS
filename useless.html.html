<head>

    <meta charset="UTF-8">

    <title>The Useless Button</title>

    <script src="https://cdn.tailwindcss.com"></script>

    <script src="https://cdnjs.cloudflare.com/ajax/libs/tone/14.7.77/Tone.js"></script>

    <link href="https://fonts.googleapis.com/css2?family=Inter:wght@400;500;700&display=swap" rel="stylesheet">

    <style>

        body {

            font-family: 'Inter', sans-serif;

            overflow: hidden; /* Prevent scrollbars when button moves */

        }

        #theButton {

            transition: all 0.3s ease-in-out;

        }

    </style>

<body id="pageBody" class="bg-slate-100 dark:bg-slate-900 transition-colors duration-500">

 

    <div class="flex flex-col items-center justify-center min-h-screen p-4 text-center">

       

        <div class="absolute top-5 right-5 text-slate-500 dark:text-slate-400">

            <p>Clicks: <span id="clickCounter" class="font-bold">0</span></p>

        </div>

 

        <div class="space-y-6">

            <h1 class="text-4xl md:text-5xl font-bold text-slate-800 dark:text-white">The Useless Button</h1>

            <p id="actionMessage" class="text-lg text-slate-600 dark:text-slate-300 min-h-[2.5em]">Click the button. See what happens. Or doesn't.</p>

           

            <div class="relative h-24 flex items-center justify-center">

                 <button id="theButton" class="absolute bg-blue-600 hover:bg-blue-700 text-white font-bold py-4 px-8 rounded-lg shadow-lg transform hover:scale-105 focus:outline-none focus:ring-4 focus:ring-blue-300 dark:focus:ring-blue-800">

                    Press Me

                </button>

            </div>

        </div>

 

    </div>

 

    <script>

        // --- DOM Element References ---

        const theButton = document.getElementById('theButton');

        const actionMessage = document.getElementById('actionMessage');

        const pageBody = document.getElementById('pageBody');

        const clickCounter = document.getElementById('clickCounter');

 

        // --- State ---

        let clicks = 0;

 

        // --- Useless Data ---

        const uselessFacts = [

            "A group of flamingos is called a flamboyance.",

            "Banging your head against a wall for one hour burns 150 calories.",

            "A shrimp's heart is in its head.",

            "It is physically impossible for pigs to look up into the sky.",

            "The 'sixth sick sheik's sixth sheep's sick' is believed to be the toughest tongue twister in the English language.",

            "A crocodile cannot stick its tongue out.",

            "On average, a person will spend about 5 years of their life waiting in lines.",

            "The national animal of Scotland is the unicorn.",

            "A single cloud can weigh more than 1 million pounds.",

            "You are now manually breathing."

        ];

 

        const buttonTexts = [

            "Again!", "Do it.", "Why?", "Pointless", "More!", "Click.", "Don't Press", "Push", "Useless"

        ];

       

        const colors = [

            'bg-slate-100 dark:bg-slate-900',

            'bg-red-100 dark:bg-red-900',

            'bg-yellow-100 dark:bg-yellow-900',

            'bg-green-100 dark:bg-green-900',

            'bg-blue-100 dark:bg-blue-900',

            'bg-indigo-100 dark:bg-indigo-900',

            'bg-purple-100 dark:bg-purple-900',

            'bg-pink-100 dark:bg-pink-900',

        ];

 

        // --- Useless Actions ---

        const uselessActions = [

            // 1. Change background color

            () => {

                const randomColorClass = colors[Math.floor(Math.random() * colors.length)];

                // Reset all possible color classes before adding the new one

                pageBody.className = '';

                colors.forEach(c => pageBody.classList.remove(...c.split(' ')));

                pageBody.classList.add(...randomColorClass.split(' '), 'transition-colors', 'duration-500');

                actionMessage.textContent = "Enjoy this new, exciting color.";

            },

            // 2. Show a useless fact

            () => {

                const fact = uselessFacts[Math.floor(Math.random() * uselessFacts.length)];

                actionMessage.textContent = fact;

            },

            // 3. Play a random beep

            () => {

                const synth = new Tone.Synth().toDestination();

                const notes = ["C4", "E4", "G4", "A4", "C5"];

                const randomNote = notes[Math.floor(Math.random() * notes.length)];

                // Use start() to ensure audio context is running

                Tone.start().then(() => {

                     synth.triggerAttackRelease(randomNote, "8n");

                });

                actionMessage.textContent = "Wasn't that a lovely sound?";

            },

            // 4. Move the button randomly

            () => {

                const x = Math.random() * (window.innerWidth - theButton.offsetWidth - 100) + 50;

                const y = Math.random() * (window.innerHeight - theButton.offsetHeight - 200) + 100; // avoid top/bottom edges

               

                // We move the parent container to keep the layout clean

                const buttonContainer = theButton.parentElement;

                buttonContainer.style.position = 'absolute';

                buttonContainer.style.left = `${x}px`;

                buttonContainer.style.top = `${y}px`;

                actionMessage.textContent = "Catch me if you can!";

            },

            // 5. Change the button's text

            () => {

                const text = buttonTexts[Math.floor(Math.random() * buttonTexts.length)];

                theButton.textContent = text;

                actionMessage.textContent = "The button has evolved.";

            },

            // 6. Do absolutely nothing

            () => {

                actionMessage.textContent = "Did something happen? Maybe. Maybe not.";

            },

            // 7. Briefly shrink the button

            () => {

                theButton.style.transform = 'scale(0.5)';

                setTimeout(() => {

                    theButton.style.transform = 'scale(1)';

                }, 200);

                actionMessage.textContent = "Woah, it got small for a second.";

            }

        ];

 

        // --- Event Listener ---

        theButton.addEventListener('click', () => {

            // Increment click counter

            clicks++;

            clickCounter.textContent = clicks;

 

            // Pick a random action and execute it

            const randomIndex = Math.floor(Math.random() * uselessActions.length);

            uselessActions[randomIndex]();

        });

 

    </script>

</body>
